import { useState, useCallback } from 'react';
import { AuditCorrections, HistoryAction } from './types'; // Adjust imports

interface UseAuditHistoryProps {
    // Callbacks to update the main UI's AuditData list
    onRestoreToActiveList: (items: AuditCorrections[]) => void;
    onRemoveFromActiveList: (items: AuditCorrections[]) => void;
    // Callback to tell Google Sheets to actually do the work
    syncToSheet: (action: HistoryAction, direction: 'undo' | 'redo') => Promise<void>;
}

export const useAuditHistory = ({ 
    onRestoreToActiveList, 
    onRemoveFromActiveList, 
    syncToSheet 
}: UseAuditHistoryProps) => {
    
    const [past, setPast] = useState<HistoryAction[]>([]);
    const [future, setFuture] = useState<HistoryAction[]>([]);
    const [isHistoryBusy, setIsHistoryBusy] = useState(false);

    /**
     * THE BOUNCER: All new actions MUST enter through here.
     */
    const record = useCallback((action: HistoryAction) => {
        setPast(prev => [...prev, action]);
        setFuture([]); // WIPE THE FUTURE (Timeline paradox prevented)
    }, []);

    /**
     * UNDO: Move from Past -> Future, Restore UI, Sync GAS
     */
    const undo = useCallback(async () => {
        if (past.length === 0 || isHistoryBusy) return;

        setIsHistoryBusy(true);
        const actionToUndo = past[past.length - 1]; // Get the top item

        try {
            // 1. Optimistic UI update: Put the cards back in the sidebar
            onRestoreToActiveList(actionToUndo.items);

            // 2. Tell GAS to put the values/highlights back
            await syncToSheet(actionToUndo, 'undo');

            // 3. Update the stacks (Past -> Future)
            setPast(prev => prev.slice(0, -1));
            setFuture(prev => [...prev, actionToUndo]);

        } catch (error) {
            // Rollback the Optimistic UI if GAS fails
            onRemoveFromActiveList(actionToUndo.items);
            console.error("Undo failed", error);
            throw error;
        } finally {
            setIsHistoryBusy(false);
        }
    }, [past, isHistoryBusy, onRestoreToActiveList, syncToSheet, onRemoveFromActiveList]);

    /**
     * REDO: Move from Future -> Past, Remove UI, Sync GAS
     */
    const redo = useCallback(async () => {
        if (future.length === 0 || isHistoryBusy) return;
        
        // YOUR TURN: 
        // Implement the reverse logic of Undo here.
        // 1. Get the top item from 'future'
        // 2. Optimistically remove the cards from the sidebar
        // 3. Tell GAS to apply the fixes again ('redo')
        // 4. Update the stacks (Future -> Past)
        // 5. Catch errors and rollback

    }, [future, isHistoryBusy, onRemoveFromActiveList, syncToSheet, onRestoreToActiveList]);

    /**
     * CLEAR: Used when the user clicks "Refresh" or loads a new sheet
     */
    const clearHistory = useCallback(() => {
        setPast([]);
        setFuture([]);
    }, []);

    return {
        record,
        undo,
        redo,
        clearHistory,
        canUndo: past.length > 0,
        canRedo: future.length > 0,
        isHistoryBusy
    };
};
