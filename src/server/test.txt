// --- UTILITIES ---

/**
 * Converts A1 notation to zero-based array indices [row, col]
 * e.g., "A1" -> { row: 0, col: 0 }, "C5" -> { row: 4, col: 2 }
 */
function a1ToIndex_(a1) {
    const match = a1.match(/([A-Z]+)(\d+)/);
    if (!match) return { row: 0, col: 0 };
    
    const colStr = match[1];
    const row = parseInt(match[2], 10) - 1;
    
    let col = 0;
    for (let i = 0; i < colStr.length; i++) {
        col = col * 26 + colStr.charCodeAt(i) - 64;
    }
    return { row, col: col - 1 };
}



// --- BULK OPERATIONS (Peak Performance) ---

// @ts-ignore
function applyFixAll(corrections) {
    if (!corrections || corrections.length === 0) return;

    const sheet = SpreadsheetApp.getActiveSheet();
    const dataRange = sheet.getDataRange();
    
    // 1. Read everything into memory (1 API Call each)
    const values = dataRange.getValues();
    
    // 2. Modify our local arrays instantly (0 API Calls)
    corrections.forEach((correction) => {
        const { row, col } = a1ToIndex_(correction.cellAddress);
        
        // Ensure our array is big enough (in case a cell is outside current bounds)
        if (values[row] !== undefined && values[row][col] !== undefined) {
            values[row][col] = correction.fixedValue;
        }
    });

    // 3. Write data back in one shot (1 API Call)
    dataRange.setValues(values);
    
    // 4. Clear UI formatting using RangeList (1 API Call)
    const addresses = corrections.map(c => c.cellAddress);
    const rangeList = sheet.getRangeList(addresses);
    rangeList.setBackground(null);
    rangeList.clearNote();
    
    SpreadsheetApp.flush(); // Force Google to apply instantly
}

// @ts-ignore
function highlightCells(corrections) {
    if (!corrections || corrections.length === 0) return;

    const sheet = SpreadsheetApp.getActiveSheet();
    const dataRange = sheet.getDataRange();
    
    // 1. Pull the Notes into memory
    const notes = dataRange.getNotes();
    
    // 2. Modify local notes array
    corrections.forEach((correction) => {
        const { row, col } = a1ToIndex_(correction.cellAddress);
        if (notes[row] !== undefined && notes[row][col] !== undefined) {
            notes[row][col] = `ISSUE ${correction.issue}\n\nFIX: ${correction.fixedValue}`;
        }
    });

    // 3. Write notes back in one shot
    dataRange.setNotes(notes);
    
    // 4. Use RangeList for colors (Lightning fast)
    const addresses = corrections.map(c => c.cellAddress);
    sheet.getRangeList(addresses).setBackground('#fce8e6');
    
    SpreadsheetApp.flush();
}







// --- SINGLE OPERATIONS ---

// @ts-ignore
function applyFix(cellAddress, fixedValue) {
    const sheet = SpreadsheetApp.getActiveSheet();
    const range = sheet.getRange(cellAddress);

    range.setValue(fixedValue);
    range.setBackground(null);
    range.clearNote();
    SpreadsheetApp.flush(); // Prevents UI desync
}

// @ts-ignore
function applyUndo(cellAddress, originalValue) {
    const sheet = SpreadsheetApp.getActiveSheet();
    const range = sheet.getRange(cellAddress);

    range.setValue(originalValue);
    SpreadsheetApp.flush();
}










// Cache and Storage
const CACHE_KEY = 'LAST_AUDIT';

// @ts-ignore
function saveAuditToCache(data) {
    try {
        // Warning: PropertiesService fails if 'data' string > 9KB.
        PropertiesService.getDocumentProperties().setProperty(CACHE_KEY, data);
    } catch (error) {
        console.error("Failed to save cache. Data might be too large.", error);
    }
}
